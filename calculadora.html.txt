<script>
let trabajos = [];

// 🔥 algoritmo optimización (usa sobrantes)
function optimizarCortes(piezas, largoBarra) {
  let barras = [];
  
  piezas.sort((a,b)=>b-a); // mayor a menor

  piezas.forEach(p => {
    let colocado = false;

    for (let i=0; i<barras.length; i++) {
      if (barras[i] >= p) {
        barras[i] -= p;
        colocado = true;
        break;
      }
    }

    if (!colocado) {
      barras.push(largoBarra - p);
    }
  });

  return {
    barras: barras.length,
    sobrante: barras.reduce((a,b)=>a+b,0)
  };
}

// 🔥 cálculo barrotes
function calcularBarrotes(ancho, separacion, perfil) {
  let espacios = Math.ceil((ancho - perfil) / (separacion + perfil));
  let barrotes = espacios - 1;
  return barrotes;
}

function agregarTrabajo() {
  let ancho = parseFloat(document.getElementById("ancho").value);
  let alto = parseFloat(document.getElementById("alto").value);
  let separacion = parseFloat(document.getElementById("separacion").value);
  let perfil = parseFloat(document.getElementById("perfil").value);
  let material = document.getElementById("material").value;
  let largoBarra = parseFloat(document.getElementById("largoBarra").value);
  let precio = parseFloat(document.getElementById("precio").value);

  if (!ancho || !alto || !precio) return;

  let piezas = [];
  let barras = 0;
  let sobrante = 0;

  if (material === "Varilla" || material === "Perfil" || material === "Ángulo" || material === "Platina") {

    let cantidad = calcularBarrotes(ancho, separacion || 10, perfil || 1);

    for (let i=0;i<cantidad;i++) {
      piezas.push(alto);
    }

    let res = optimizarCortes(piezas, largoBarra);
    barras = res.barras;
    sobrante = res.sobrante;

  } else {
    // láminas
    let area = ancho * alto;
    let lamina = 122 * 244; // cm aprox

    barras = Math.ceil((area / lamina) * 1.10);
    sobrante = 0;
  }

  let subtotal = barras * precio;

  trabajos.push({
    material,
    barras,
    sobrante,
    subtotal
  });

  mostrar();
}

function mostrar() {
  let lista = document.getElementById("lista");
  let resultado = document.getElementById("resultado");

  lista.innerHTML = "";
  let total = 0;

  trabajos.forEach((t,i)=>{
    lista.innerHTML += `
      <div>
        ${i+1}. ${t.material} | Barras: ${t.barras} | Sobrante: ${t.sobrante.toFixed(2)}
      </div>
    `;
    total += t.subtotal;
  });

  resultado.innerHTML = `
    Material: ${total.toFixed(2)} <br>
    Precio final: ${(total * 2.5).toFixed(2)}
  `;
}
