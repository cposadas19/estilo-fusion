<script>
let db = []; let proyectoActual = [];

function ajustarCampos() {
    const ofi = document.getElementById('p_oficio').value;
    document.getElementById('detalles_metal').style.display = ofi === 'metal' ? 'block' : 'none';
    document.getElementById('detalles_pvc_mueble').style.display = (ofi === 'pvc' || ofi === 'mueble') ? 'block' : 'none';
    document.getElementById('detalles_elec').style.display = ofi === 'electricidad' ? 'block' : 'none';
    document.getElementById('campos_medidas').style.display = ofi === 'electricidad' ? 'none' : 'block';
}

function guardarMaterial() {
    const material = { id: Date.now(), nombre: document.getElementById('inv_nombre').value, tipo: document.getElementById('inv_tipo').value, precio: parseFloat(document.getElementById('inv_precio').value), unidad: parseInt(document.getElementById('inv_unidad').value) };
    if(!material.nombre || isNaN(material.precio)) return alert("Completa los datos");
    db.push(material);
    actualizarInterfaz();
    document.getElementById('inv_nombre').value = "";
}

function actualizarInterfaz() {
    const sM = document.getElementById('sel_marco'), sB = document.getElementById('sel_barrote'), sL = document.getElementById('sel_lamina');
    sM.innerHTML = sB.innerHTML = sL.innerHTML = '<option value="">Seleccione material...</option>';
    db.forEach(m => {
        const opt = `<option value="${m.id}">${m.nombre}</option>`;
        if(m.tipo === 'Marco') sM.innerHTML += opt;
        else if(m.tipo === 'Barrote') sB.innerHTML += opt;
        else if(m.tipo === 'Lamina') sL.innerHTML += opt;
    });
    const lI = document.getElementById('lista_inv'); lI.innerHTML = "";
    db.forEach((m, i) => lI.innerHTML += `<div class="lista-item" style="font-size:12px;">${m.nombre} (L.${m.precio}) <button class="btn-del" onclick="db.splice(${i},1);actualizarInterfaz();">X</button></div>`);
}

function agregarAlProyecto() {
    const ofi = document.getElementById('p_oficio').value;
    const desc = document.getElementById('p_desc').value || "Trabajo";
    const ganP = parseFloat(document.getElementById('p_gan').value) / 100;
    const mo = parseFloat(document.getElementById('p_mo').value) || 0;
    let costoMat = 0, detalle = "";

    try {
        if(ofi === 'metal') {
            const m = db.find(x => x.id == document.getElementById('sel_marco').value);
            const b = db.find(x => x.id == document.getElementById('sel_barrote').value);
            const a = parseFloat(document.getElementById('p_ancho').value), h = parseFloat(document.getElementById('p_alto').value);
            const gr = parseFloat(document.getElementById('p_grosor').value), sep = parseFloat(document.getElementById('p_sep').value);
            const nEsp = Math.round(a/sep), cantB = nEsp-1, corteB = h-(gr*2);
            costoMat = (Math.ceil(((a*2)+(h*2))/12/20)*m.precio) + (Math.ceil((corteB*cantB)/12/30)*b.precio);
            detalle = `${cantB} barrotes de ${corteB.toFixed(2)}"`;
        } else if(ofi === 'pvc' || ofi === 'mueble') {
            const l = db.find(x => x.id == document.getElementById('sel_lamina').value);
            const a = parseFloat(document.getElementById('p_ancho').value), h = parseFloat(document.getElementById('p_alto').value);
            const extra = parseFloat(document.getElementById('p_extra').value);
            costoMat = (ofi === 'pvc') ? (Math.ceil(((a*4)+(h*4))*0.0254/6)*l.precio) + extra : (Math.ceil((a*h)/(48*96))*l.precio) + extra;
            detalle = `${ofi.toUpperCase()}: ${a}x${h}"`;
        } else {
            costoMat = parseInt(document.getElementById('p_cant_elec').value) * parseFloat(document.getElementById('p_precio_elec').value);
            detalle = `${document.getElementById('p_cant_elec').value} puntos eléctricos`;
        }

        const inversion = costoMat + mo;
        proyectoActual.push({ desc, inversion: costoMat, mo, ganancia: inversion * ganP, detalle });
        renderProyecto();
    } catch(e) { alert("Error: Selecciona los materiales primero"); }
}

function renderProyecto() {
    const cont = document.getElementById('lista_proyecto'); cont.innerHTML = "";
    let tInv = 0, tMo = 0, tGan = 0;
    proyectoActual.forEach((p, i) => {
        tInv += p.inversion; tMo += p.mo; tGan += p.ganancia;
        cont.innerHTML += `<div class="lista-item"><strong>${p.desc}</strong><br><small style="color:#aaa">${p.detalle}</small><br>Venta: L. ${(p.inversion+p.mo+p.ganancia).toLocaleString()}<button class="btn-del" onclick="proyectoActual.splice(${i},1);renderProyecto();">X</button></div>`;
    });
    document.getElementById('res_final').style.display = proyectoActual.length > 0 ? 'block' : 'none';
    document.getElementById('detalle_costos').innerHTML = `<div class="res-line"><span>Costo Materiales:</span> <span>L. ${tInv.toLocaleString()}</span></div><div class="res-line"><span>Mano de Obra:</span> <span>L. ${tMo.toLocaleString()}</span></div><div class="res-line"><span style="color:var(--dorado)">Utilidad Libre:</span> <span style="font-weight:bold; color:var(--dorado)">L. ${tGan.toLocaleString()}</span></div>`;
    document.getElementById('gran_total').innerText = "PRECIO TOTAL: L. " + (tInv+tMo+tGan).toLocaleString();
}

function openTab(evt, name) {
    let card = document.getElementsByClassName("card");
    for (let i = 0; i < card.length; i++) card[i].style.display = "none";
    let tab = document.getElementsByClassName("tab");
    for (let i = 0; i < tab.length; i++) tab[i].classList.remove("active");
    document.getElementById(name).style.display = "block";
    evt.currentTarget.classList.add("active");
}
