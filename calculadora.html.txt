<script>
// --- BASE DE DATOS PERSISTENTE ---
let mats = JSON.parse(localStorage.getItem('mats_v2')) || [
    {n: "Tubo 1 cal. 18 (6 mts)", p: 230},
    {n: "Varilla Lisa 3/8 (9 mts)", p: 165},
    {n: "Ángulo 1 x 1/8 (6 mts)", p: 175},
    {n: "Disco de Corte 14", p: 235}
];

let itemsProyecto = [];

function cargarPagina() {
    document.getElementById('meta_input').value = localStorage.getItem('meta_diaria') || "";
    renderDB();
    actualizarSelects();
}

function renderDB() {
    const c = document.getElementById('cuerpo-mats');
    c.innerHTML = "";
    mats.forEach(m => {
        const tr = document.createElement('tr');
        tr.innerHTML = `<td><input type="text" class="n" value="${m.n}"></td><td><input type="number" class="p" value="${m.p}"></td>`;
        c.appendChild(tr);
    });
}

function addFila(id) {
    const tr = document.createElement('tr');
    tr.innerHTML = `<td><input type="text" class="n"></td><td><input type="number" class="p"></td>`;
    document.getElementById(id).appendChild(tr);
}

function guardarDB() {
    mats = Array.from(document.querySelectorAll('#cuerpo-mats tr')).map(f => ({
        n: f.querySelector('.n').value, p: parseFloat(f.querySelector('.p').value) || 0
    }));
    localStorage.setItem('mats_v2', JSON.stringify(mats));
    actualizarSelects();
    alert("Precios actualizados.");
}

function actualizarSelects() {
    const options = mats.map((m, i) => `<option value="${i}">${m.n}</option>`).join('');
    document.getElementById('v_marco_select').innerHTML = options;
    document.getElementById('v_barrote_select').innerHTML = options;
}

function guardarMeta() {
    localStorage.setItem('meta_diaria', document.getElementById('meta_input').value);
}

// --- CÁLCULOS ---
function añadirVerja() {
    const idMarco = document.getElementById('v_marco_select').value;
    const idBarrote = document.getElementById('v_barrote_select').value;
    
    const item = {
        ancho: parseFloat(document.getElementById('v_ancho').value),
        alto: parseFloat(document.getElementById('v_alto').value),
        marco: mats[idMarco],
        barrote: mats[idBarrote],
        sepDeseada: parseFloat(document.getElementById('v_sep').value),
        mo: parseFloat(document.getElementById('v_mo').value),
        ganancia: parseFloat(document.getElementById('v_utilidad').value)
    };

    itemsProyecto.push(item);
    document.getElementById('resumen-total').style.display = 'block';
    renderResumen();
}

function renderResumen() {
    let totalPiesMaterial = 0;
    let costoTotalMaterial = 0;
    let sumaManoObra = 0;
    let htmlItems = "";

    itemsProyecto.forEach((v, i) => {
        // Lógica de barrotes e igualación de espacios
        const gBarrote = 1; // Pulgada de espesor promedio
        const anchoUtil = v.ancho - 2; // Menos el marco
        const numEspacios = Math.round(anchoUtil / (v.sepDeseada + gBarrote));
        const numBarrotes = numEspacios - 1;
        const sepReal = (anchoUtil - (numBarrotes * gBarrote)) / numEspacios;
        const largoBarrote = v.alto - 2;

        const piesBarrotes = (numBarrotes * largoBarrote) / 12;
        const piesMarco = ((v.ancho * 2) + (v.alto * 2)) / 12;

        totalPiesMaterial += (piesBarrotes + piesMarco);
        sumaManoObra += v.mo;
        
        // Calcular costo proporcional basado en el precio del perfil de 6mts (aprox 20 pies)
        costoTotalMaterial += ( (piesBarrotes * v.barrote.p) / 19.6 ) + ( (piesMarco * v.marco.p) / 19.6 );

        htmlItems += `<b>Verja ${i+1}:</b> ${numBarrotes} pzas de ${largoBarrote}" | Espacio: ${sepReal.toFixed(2)}"<br>`;
    });

    const perfilesComprar = Math.ceil(totalPiesMaterial / 19.6);
    
    document.getElementById('lista_items').innerHTML = htmlItems;
    
    document.getElementById('calculo_materiales').innerHTML = `
        📐 <b>CÁLCULO FÍSICO:</b><br>
        Material total: ${totalPiesMaterial.toFixed(1)} pies lineales.<br>
        <b>Comprar: ${perfilesComprar} perfiles aprox.</b>
    `;

    // FÓRMULA FINANCIERA: (Costo Mat + MO) / (1 - %Ganancia)
    const costoBase = costoTotalMaterial + sumaManoObra;
    const porcentaje = itemsProyecto[0].ganancia / 100;
    const precioVenta = costoBase / (1 - porcentaje);
    const gananciaReal = precioVenta - costoBase;

    document.getElementById('calculo_financial').innerHTML = `
        💰 <b>RESULTADO ECONÓMICO:</b><br>
        Costo Materiales: L. ${costoTotalMaterial.toFixed(2)}<br>
        Suma Mano Obra: L. ${sumaManoObra.toFixed(2)}<br>
        ---<br>
        <span style="font-size:1.2em;">PRECIO A COBRAR: <b>L. ${precioVenta.toFixed(2)}</b></span><br>
        <span style="color:green;">TU GANANCIA LIMPIA: L. ${gananciaReal.toFixed(2)}</span>
    `;
}

function nuevoProyecto() {
    if(confirm("¿Borrar todo para un nuevo cliente?")) {
        itemsProyecto = [];
        document.getElementById('resumen-total').style.display = 'none';
    }
}

function openTab(id) {
    document.querySelectorAll('.tab-content').forEach(t => t.style.display = 'none');
    document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
    document.getElementById(id).style.display = 'block';
    event.currentTarget.classList.add('active');
}

window.onload = cargarPagina;
