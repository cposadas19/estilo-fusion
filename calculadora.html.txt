<script>
let db = JSON.parse(localStorage.getItem('fusion_db')) || [];
let proyectoActual = [];

function guardarMaterial() {
    const material = {
        id: Date.now(),
        nombre: document.getElementById('inv_nombre').value,
        precio: parseFloat(document.getElementById('inv_precio').value),
        unidad: parseInt(document.getElementById('inv_unidad').value)
    };
    if(!material.nombre || isNaN(material.precio)) return alert("Pon nombre y precio");
    db.push(material);
    localStorage.setItem('fusion_db', JSON.stringify(db));
    actualizarInterfaz();
    document.getElementById('inv_nombre').value = "";
}

function actualizarInterfaz() {
    const sMarco = document.getElementById('sel_marco');
    const sBarro = document.getElementById('sel_barrote');
    const sCons = document.getElementById('sel_cons');
    
    sMarco.innerHTML = sBarro.innerHTML = sCons.innerHTML = '<option value="">-- Selecciona --</option>';
    
    db.forEach(m => {
        const opt = `<option value="${m.id}">${m.nombre} (L. ${m.precio})</option>`;
        sMarco.innerHTML += opt;
        sBarro.innerHTML += opt;
        sCons.innerHTML += opt;
    });

    const lInv = document.getElementById('lista_inv');
    lInv.innerHTML = "<h4>Materiales en Base de Datos:</h4>";
    db.forEach((m, index) => {
        lInv.innerHTML += `<div class="lista-item">${m.nombre} (L. ${m.precio}) <button class="btn-del" onclick="db.splice(${index},1);localStorage.setItem('fusion_db', JSON.stringify(db));actualizarInterfaz();">X</button></div>`;
    });
}

function agregarAlProyecto() {
    const idM = document.getElementById('sel_marco').value;
    const idB = document.getElementById('sel_barrote').value;
    const idC = document.getElementById('sel_cons').value;
    
    const m = db.find(x => x.id == idM);
    const b = db.find(x => x.id == idB);
    const c = db.find(x => x.id == idC);
    
    const a = parseFloat(document.getElementById('p_ancho').value);
    const h = parseFloat(document.getElementById('p_alto').value);
    const grosor = parseFloat(document.getElementById('p_grosor').value);
    const sep = parseFloat(document.getElementById('p_sep').value);
    const mo = parseFloat(document.getElementById('p_mo').value) || 0;
    const ganP = parseFloat(document.getElementById('p_gan').value) / 100;

    let costoMateriales = 0;
    
    if(m) costoMateriales += Math.ceil(((a*2)+(h*2))/12/m.unidad) * m.precio;
    
    if(b) {
        const numEsp = Math.round(a / sep);
        const cantB = numEsp - 1;
        const corteB = h - (grosor * 2);
        costoMateriales += Math.ceil((corteB * cantB)/12/b.unidad) * b.precio;
    }
    
    if(c) costoMateriales += c.precio;

    const inversion = costoMateriales + mo;
    const precioVenta = inversion + (inversion * ganP);

    proyectoActual.push({
        desc: document.getElementById('p_desc').value || "Pieza",
        costo: precioVenta
    });
    renderProyecto();
}

function renderProyecto() {
    const cont = document.getElementById('lista_proyecto');
    cont.innerHTML = "";
    let total = 0;
    proyectoActual.forEach((p, i) => {
        total += p.costo;
        cont.innerHTML += `<div class="lista-item"><strong>${p.desc}</strong><br>Precio: L. ${p.costo.toLocaleString()}<button class="btn-del" onclick="proyectoActual.splice(${i},1);renderProyecto();">X</button></div>`;
    });
    document.getElementById('res_final').style.display = proyectoActual.length > 0 ? 'block' : 'none';
    document.getElementById('gran_total').innerText = "TOTAL OBRA: L. " + total.toLocaleString();
}

function openTab(evt, name) {
    let card = document.getElementsByClassName("card");
    for (let i = 0; i < card.length; i++) card[i].style.display = "none";
    let tab = document.getElementsByClassName("tab");
    for (let i = 0; i < tab.length; i++) tab[i].classList.remove("active");
    document.getElementById(name).style.display = "block";
    evt.currentTarget.classList.add("active");
}

actualizarInterfaz();
</script>
