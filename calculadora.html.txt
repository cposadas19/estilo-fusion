    <script>
        // Lista completa de materiales con sus grosores aproximados (en pulgadas)
        const db_inicial = [
            { nom: "Tubo 1/2 Cal. 18", pre: 160, grosor: 0.5 },
            { nom: "Tubo 1 Cal. 18", pre: 230, grosor: 1.0 },
            { nom: "Tubo 1 1/2 Cal. 14", pre: 600, grosor: 1.5 },
            { nom: "Angulo 1 x 1/8", pre: 175, grosor: 1.0 },
            { nom: "Angulo 1 1/2 x 1/8", pre: 285, grosor: 1.5 },
            { nom: "Varilla Lisa 3/8", pre: 165, grosor: 0.375 },
            { nom: "Varilla Lisa 1/2", pre: 275, grosor: 0.5 }
        ];

        function showTab(id) {
            document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
            document.querySelectorAll('.tab-link').forEach(l => l.classList.remove('active-tab'));
            document.getElementById(id).classList.add('active');
            document.getElementById('btn-' + id).classList.add('active-tab');
            if(id === 'metal') actualizarSelectores();
            if(id === 'db') renderDB();
        }

        function actualizarSelectores() {
            const marcos = document.getElementById('m-material-marco');
            const barrotes = document.getElementById('m-material-barrote');
            const options = db_inicial.map((m, index) => `<option value="${index}">${m.nom}</option>`).join('');
            marcos.innerHTML = options;
            barrotes.innerHTML = options;
        }

        function calcularTodo() {
            const ancho = parseFloat(document.getElementById('m-ancho').value);
            const alto = parseFloat(document.getElementById('m-alto').value);
            const sepDeseada = parseFloat(document.getElementById('m-sep').value);
            const sobranteExistente = parseFloat(document.getElementById('m-sobrante').value) || 0;
            
            const matMarco = db_inicial[document.getElementById('m-material-marco').value];
            const matBarrote = db_inicial[document.getElementById('m-material-barrote').value];

            // 1. CÁLCULO DE MEDIDA DE BARROTE (Alto - Grosor de Marco arriba y abajo)
            const largoBarrote = alto - (matMarco.grosor * 2);

            // 2. CÁLCULO DE ESPACIOS IGUALES
            const anchoInterno = ancho - (matMarco.grosor * 2);
            const cantidadBarrotes = Math.round(anchoInterno / (sepDeseada + matBarrote.grosor));
            const sepReal = (anchoInterno - (cantidadBarrotes * matBarrote.grosor)) / (cantidadBarrotes + 1);

            // 3. MATERIAL TOTAL Y SOBRANTES
            const totalPulgadasNecesarias = cantidadBarrotes * largoBarrote;
            const pulgadasRealesAComprar = totalPulgadasNecesarias - sobranteExistente;
            const tiras = Math.ceil((pulgadasRealesAComprar / 12) / 19.6);
            const nuevoSobrante = (tiras * 19.6 * 12) - pulgadasRealesAComprar;

            const res = document.getElementById('resultado');
            res.classList.remove('hidden');
            res.innerHTML = `
                <div class="border-b border-slate-600 pb-2 mb-2">
                    <p class="text-yellow-400 font-black uppercase text-[10px] tracking-widest">Manual de Taller</p>
                    <p class="text-lg">🔪 Cortar <span class="font-black">${cantidadBarrotes}</span> barrotes de <span class="font-black text-yellow-400">${largoBarrote.toFixed(2)}"</span></p>
                    <p class="text-sm">↔️ Separación exacta: <span class="font-bold">${sepReal.toFixed(2)}"</span></p>
                </div>
                <div class="text-[10px] space-y-1">
                    <p>📦 Comprar: <span class="text-white font-bold">${tiras > 0 ? tiras : 0} tiras</span> de ${matBarrote.nom}</p>
                    <p class="text-green-400">♻️ Usaste ${sobranteExistente}" de sobrante anterior.</p>
                    <p class="text-red-400 font-bold">⚠️ Te sobrarán ${nuevoSobrante.toFixed(1)}" para el próximo proyecto.</p>
                </div>
            `;
        }

        function renderDB() {
            document.getElementById('tabla-db').innerHTML = db_inicial.map(m => `
                <div class="flex justify-between p-2 border-b bg-gray-50 uppercase font-bold text-slate-700">
                    <span>${m.nom}</span> <span>L ${m.pre}</span>
                </div>
            `).join('');
        }

        window.onload = () => { showTab('metal'); };
    </script>
