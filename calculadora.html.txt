<script>
let db = JSON.parse(localStorage.getItem('fusion_db')) || [];
let proyecto = [], carritoConsumibles = [];

function openTab(e, n) {
    document.querySelectorAll('.card').forEach(x => x.style.display = 'none');
    document.querySelectorAll('.tab').forEach(x => x.classList.remove('active'));
    document.getElementById(n).style.display = 'block'; e.currentTarget.classList.add('active');
}

function guardarDB() {
    const n = document.getElementById('db_n').value, p = parseFloat(document.getElementById('db_p').value);
    if(!n || isNaN(p)) return;
    db.push({ id: Date.now(), n, p });
    localStorage.setItem('fusion_db', JSON.stringify(db));
    document.getElementById('db_n').value = ""; document.getElementById('db_p').value = "";
    actualizarSelects();
}

function actualizarSelects() {
    const ids = ['sel_marco', 'sel_barro', 'sel_cons'];
    ids.forEach(id => {
        const s = document.getElementById(id); s.innerHTML = '<option value="">--Seleccionar--</option>';
        db.forEach(m => s.innerHTML += `<option value="${m.id}">${m.n}</option>`);
    });
    const l = document.getElementById('db_lista_visual'); l.innerHTML = "";
    db.forEach((m, i) => l.innerHTML += `<div class="lista-item">${m.n} (L.${m.p}) <button class="btn-del" onclick="db.splice(${i},1);localStorage.setItem('fusion_db',JSON.stringify(db));actualizarSelects();">X</button></div>`);
}

function sumarConsumible() {
    const id = document.getElementById('sel_cons').value;
    const cant = parseFloat(document.getElementById('cons_cant').value) || 1;
    const m = db.find(x => x.id == id);
    if(m) {
        carritoConsumibles.push({ ...m, cantidadDeseada: cant });
        renderBadges();
    }
}

function renderBadges() {
    const c = document.getElementById('lista_badges_cons'); c.innerHTML = "";
    carritoConsumibles.forEach((x, i) => c.innerHTML += `<span class="mini-badge">${x.cantidadDeseada}x ${x.n} <span onclick="carritoConsumibles.splice(${i},1);renderBadges();" style="color:red; cursor:pointer; margin-left:5px;">x</span></span>`);
}

function agregarAlProyecto() {
    const idM = document.getElementById('sel_marco').value;
    const idB = document.getElementById('sel_barro').value;
    const m = db.find(x => x.id == idM);
    const b = db.find(x => x.id == idB);
    const a = parseFloat(document.getElementById('p_ancho').value), h = parseFloat(document.getElementById('p_alto').value);
    const gr = parseFloat(document.getElementById('p_grosor').value), sep = parseFloat(document.getElementById('p_sep').value);
    const mo = parseFloat(document.getElementById('p_mo').value), gan = parseFloat(document.getElementById('p_gan').value);
    const uM = parseInt(document.getElementById('p_unit_m').value), uB = parseInt(document.getElementById('p_unit_b').value);

    if(!m) return alert("Selecciona material para marco");

    const nEsp = Math.round(a/sep), cantB = nEsp-1, corteB = h-(gr*2);
    
    proyecto.push({
        desc: document.getElementById('p_desc').value || "Pieza",
        mo, gan,
        mats: [
            { id: m.id, n: m.n, p: m.p, u: uM, pies: ((a*2)+(h*2))/12 },
            { id: b ? b.id : null, n: b ? b.n : '', p: b ? b.p : 0, u: uB, pies: b ? (corteB*cantB)/12 : 0 }
        ],
        cons: [...carritoConsumibles]
    });

    carritoConsumibles = []; renderBadges(); renderProyecto();
}

function renderProyecto() {
    const l = document.getElementById('lista_proyecto_visual'); l.innerHTML = "";
    let matResumen = {}, totalMO = 0, totalCons = 0;

    proyecto.forEach((p, i) => {
        totalMO += p.mo;
        p.mats.forEach(m => {
            if(m.id) {
                if(!matResumen[m.id]) matResumen[m.id] = { n: m.n, p: m.p, u: m.u, piesTotales: 0 };
                matResumen[m.id].piesTotales += m.pies;
            }
        });
        p.cons.forEach(c => totalCons += (c.p * c.cantidadDeseada));
        l.innerHTML += `<div class="lista-item"><strong>${p.desc}</strong> <button class="btn-del" onclick="proyecto.splice(${i},1);renderProyecto();">X</button></div>`;
    });

    let totalCostoMaterial = 0, htmlCompras = "🛒 <b>Comprar:</b><br>";
    for (let id in matResumen) {
        const m = matResumen[id];
        const tiras = Math.ceil(m.piesTotales / m.u);
        totalCostoMaterial += (tiras * m.p);
        htmlCompras += `• ${m.n}: ${tiras} tiras<br>`;
    }

    const subtotal = totalCostoMaterial + totalCons + totalMO;
    const precioFinal = subtotal + (subtotal * (30/100)); // Usando 30% como ejemplo sugerido

    document.getElementById('res_final').style.display = proyecto.length > 0 ? 'block' : 'none';
    document.getElementById('detalle_compras').innerHTML = htmlCompras;
    document.getElementById('res_det_precios').innerHTML = `
        <div style="display:flex; justify-content:space-between; font-size:14px;"><span>Hierro/Tiras:</span><span>L. ${totalCostoMaterial.toLocaleString()}</span></div>
        <div style="display:flex; justify-content:space-between; font-size:14px;"><span>Insumos Totales:</span><span>L. ${totalCons.toLocaleString()}</span></div>
        <div style="display:flex; justify-content:space-between; font-size:14px;"><span>Mano Obra:</span><span>L. ${totalMO.toLocaleString()}</span></div>
    `;
    document.getElementById('gran_total').innerText = "TOTAL VENTA: L. " + precioFinal.toLocaleString();
}
actualizarSelects();
</script>
