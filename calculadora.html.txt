// --- LÓGICA DE BASE DE DATOS ---
let db = JSON.parse(localStorage.getItem('fusion_db')) || [];
let presupuesto = [];

function guardarEnBaseDeDatos() {
    const nuevo = {
        id: Date.now(),
        nombre: document.getElementById('inv_nombre').value,
        tipo: document.getElementById('inv_tipo').value,
        precio: parseFloat(document.getElementById('inv_precio').value),
        grosor: parseFloat(document.getElementById('inv_grosor').value) || 0,
        unidad: parseInt(document.getElementById('inv_unidad').value)
    };
    db.push(nuevo);
    localStorage.setItem('fusion_db', JSON.stringify(db));
    alert("Material guardado correctamente");
    actualizarSelectores();
    renderInventario();
}

function actualizarSelectores() {
    const marcoS = document.getElementById('sel_marco');
    const barroteS = document.getElementById('sel_barrote');
    const consS = document.getElementById('sel_consumible');
    
    marcoS.innerHTML = '<option value="">Seleccione Marco...</option>';
    barroteS.innerHTML = '<option value="">Seleccione Barrote...</option>';
    consS.innerHTML = '<option value="">Seleccione Consumible...</option>';

    db.forEach(m => {
        const opt = `<option value="${m.id}">${m.nombre} - L. ${m.precio}</option>`;
        if(m.tipo === 'Marco') marcoS.innerHTML += opt;
        if(m.tipo === 'Barrote') barroteS.innerHTML += opt;
        if(m.tipo === 'Consumible') consS.innerHTML += opt;
    });
}

function renderInventario() {
    const cont = document.getElementById('lista_inventario');
    cont.innerHTML = "<h4>Materiales Registrados:</h4>";
    db.forEach((m, i) => {
        cont.innerHTML += `<div class="lista-item">${m.nombre} (${m.tipo}) <button class="btn-del" onclick="borrarDB(${i})">Eliminar</button></div>`;
    });
}

function borrarDB(i) {
    db.splice(i, 1);
    localStorage.setItem('fusion_db', JSON.stringify(db));
    actualizarSelectores();
    renderInventario();
}

// --- LÓGICA DE COTIZACIÓN ---
function agregarProyecto() {
    const idMarco = document.getElementById('sel_marco').value;
    const idBarrote = document.getElementById('sel_barrote').value;
    
    if(!idMarco) return alert("Selecciona un material de marco de tu base de datos");

    const m = db.find(x => x.id == idMarco);
    const b = db.find(x => x.id == idBarrote);
    
    const a = parseFloat(document.getElementById('job_ancho').value);
    const h = parseFloat(document.getElementById('job_alto').value);
    const sep = parseFloat(document.getElementById('job_sep').value);

    // Lógica técnica barrotes
    const numEsp = Math.round(a / sep);
    const cantB = numEsp - 1;
    const corteB = h - (m.grosor * 2);
    
    const piesMarco = ((a*2)+(h*2))/12;
    const piesB = b ? (corteB * cantB)/12 : 0;
    
    const costoM = Math.ceil(piesMarco/m.unidad) * m.precio;
    const costoB = b ? Math.ceil(piesB/b.unidad) * b.precio : 0;
    const mo = parseFloat(document.getElementById('job_mo').value) || 0;

    presupuesto.push({
        desc: document.getElementById('job_desc').value,
        costo: costoM + costoB + mo,
        detalles: `Marco: ${m.nombre} | Barrotes: ${cantB} piezas de ${corteB}"`
    });
    renderCotizacion();
}

function renderCotizacion() {
    const cont = document.getElementById('lista_cotizacion');
    cont.innerHTML = "";
    let total = 0;
    presupuesto.forEach((p, i) => {
        total += p.costo;
        cont.innerHTML += `<div class="lista-item"><strong>${p.desc}</strong><br>${p.detalles}<br>Subtotal: L. ${p.costo}<button class="btn-del" onclick="presupuesto.splice(${i},1);renderCotizacion();">X</button></div>`;
    });
    document.getElementById('resultado_final').style.display = presupuesto.length > 0 ? 'block' : 'none';
    document.getElementById('total_final').innerText = "TOTAL: L. " + total.toLocaleString();
}

function openTab(evt, name) {
    let card = document.getElementsByClassName("card");
    for (let i = 0; i < card.length; i++) card[i].classList.remove("active");
    let tab = document.getElementsByClassName("tab");
    for (let i = 0; i < tab.length; i++) tab[i].classList.remove("active");
    document.getElementById(name).classList.add("active");
    evt.currentTarget.classList.add("active");
}

// Iniciar
actualizarSelectores();
renderInventario();
