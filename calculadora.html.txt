        // LISTA DE MATERIALES COMPLETA (TU LISTA)
        const materialesIniciales = [

            { id:13, nom: "Plywood 1/4 (6mm)", pre: 695, cat: "madera" },
            { id:14, nom: "Plywood 3/4 (18mm)", pre: 1625, cat: "madera" },
            { id:15, nom: "Galón Anticorrosiva", pre: 1050, cat: "insumo" },
            { id:16, nom: "Galón Esmalte", pre: 1165, cat: "insumo" },
            { id:17, nom: "Galón de Thinner", pre: 245, cat: "insumo" },
            { id:18, nom: "Electrodo 1/8 lb", pre: 68, cat: "insumo" },
            { id:19, nom: "Disco de Corte 14 Dewalt", pre: 235, cat: "insumo" },
            { id:20, nom: "Disco de Pulir (Lija G60)", pre: 55, cat: "insumo" }
        ];

        let db = JSON.parse(localStorage.getItem('cotizador_db')) || materialesIniciales;

        function showTab(id) {
            document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
            document.querySelectorAll('.tab-link').forEach(l => l.classList.remove('active-tab'));
            document.getElementById(id).classList.add('active');
            document.getElementById('btn-' + id).classList.add('active-tab');
            renderDB();
            actualizarSelectores();
        }

        function renderDB() {
            const t = document.getElementById('tabla-db');
            t.innerHTML = db.sort((a,b) => a.nom.localeCompare(b.nom)).map(i => `
                <tr class="hover:bg-gray-50">
                    <td class="p-2 border-b">${i.nom}<br><span class="text-[8px] bg-gray-100 px-1 text-gray-500 rounded font-bold uppercase">${i.cat}</span></td>
                    <td class="p-2 border-b text-right font-bold text-green-700">L ${i.pre.toLocaleString()}</td>
                    <td class="p-2 border-b text-center"><button onclick="borrar(${i.id})" class="text-red-500 font-black text-lg">×</button></td>
                </tr>
            `).join('');
            localStorage.setItem('cotizador_db', JSON.stringify(db));
        }

        function addDB() {
            const nom = document.getElementById('db-nom').value;
            const pre = parseFloat(document.getElementById('db-pre').value);
            const cat = document.getElementById('db-cat').value;
            if(nom && pre) {
                db.push({ id: Date.now(), nom, pre, cat });
                document.getElementById('db-nom').value = '';
                document.getElementById('db-pre').value = '';
                renderDB();
                actualizarSelectores();
            }
        }

        function borrar(id) {
            if(confirm('¿Eliminar este material?')) {
                db = db.filter(i => i.id !== id);
                renderDB();
                actualizarSelectores();
            }
        }

        function actualizarSelectores() {
            document.querySelectorAll('.select-material').forEach(s => {
                s.innerHTML = db.map(i => `<option value="${i.pre}">${i.nom}</option>`).join('');
            });
        }

        function calcularMetal() {
            const ancho = parseFloat(document.getElementById('m-ancho').value);
            const alto = parseFloat(document.getElementById('m-alto').value);
            const sepDeseada = parseFloat(document.getElementById('m-sep').value);
            const mo = parseFloat(document.getElementById('m-mo').value);
            const gananciaPct = parseFloat(document.getElementById('m-ganancia').value) / 100;
            const preMaterial = parseFloat(document.getElementById('m-material').value);

            // LOGICA: Altura barrote (-2" por marcos)
            const largoBarrote = alto - 2;
            
            // LOGICA: Espacios iguales (asumiendo perfil de 1")
            const cantidadBarrotes = Math.round((ancho - 2) / (sepDeseada + 1));
            const sepReal = (ancho - 2 - (cantidadBarrotes * 1)) / (cantidadBarrotes + 1);

            // Cálculo económico aproximado
            const piesLineales = (cantidadBarrotes * largoBarrote) / 12;
            const costoMaterial = piesLineales * (preMaterial / 19.6); // Estimación por pie
            const subtotal = costoMaterial + mo;
            const totalFinal = subtotal / (1 - gananciaPct);

            const res = document.getElementById('res-metal');
            res.classList.remove('hidden');
            res.innerHTML = `
                <div class="border-b border-slate-600 pb-3 mb-3">
                    <p class="text-[10px] text-yellow-400 font-bold uppercase tracking-widest">Medidas de Taller</p>
                    <div class="grid grid-cols-2 gap-2 mt-1">
                        <p>Corte Barrote: <span class="font-bold text-lg">${largoBarrote}"</span></p>
                        <p>Cantidad: <span class="font-bold text-lg">${cantidadBarrotes} pzas</span></p>
                        <p class="col-span-2">Espacio entre Barrotes: <span class="text-yellow-400 font-bold">${sepReal.toFixed(2)}"</span></p>
                    </div>
                </div>
                <div class="flex justify-between items-end">
                    <div>
                        <p class="text-[10px] text-gray-400 uppercase">Costo Mat + MO</p>
                        <p class="text-sm">L ${subtotal.toFixed(2)}</p>
                    </div>
                    <div class="text-right">
                        <p class="text-[10px] text-green-400 uppercase font-bold">Total a Cobrar</p>
                        <p class="text-2xl font-black text-yellow-400">L ${totalFinal.toFixed(2)}</p>
                    </div>
                </div>
                <p class="text-center text-[9px] text-green-400 pt-2 border-t border-slate-700 uppercase">Tu Ganancia Neta: L ${(totalFinal - subtotal).toFixed(2)}</p>
            `;
        }

        window.onload = () => { renderDB(); actualizarSelectores(); };
    </script>
