<script>

let db = JSON.parse(localStorage.getItem('fusion_db')) || [];
let proyecto = [], carritoConsumibles = [], carritoPVC = [];

function openTab(e, n) {
    document.querySelectorAll('.card').forEach(x => x.style.display = 'none');
    document.querySelectorAll('.tab').forEach(x => x.classList.remove('active'));
    document.getElementById(n).style.display = 'block'; e.currentTarget.classList.add('active');
}

function guardarDB() {
    const n = document.getElementById('db_n').value, p = parseFloat(document.getElementById('db_p').value);
    if(!n || isNaN(p)) return;
    db.push({ id: Date.now(), n, p });
    localStorage.setItem('fusion_db', JSON.stringify(db));
    document.getElementById('db_n').value = ""; document.getElementById('db_p').value = "";
    actualizarSelects();
}

function actualizarSelects() {
    const ids = ['sel_marco','sel_barrote','sel_cons','pv_marco','pv_hoja','pv_cons'];
    ids.forEach(id => {
        const s = document.getElementById(id);
        if(!s) return;
        s.innerHTML = '<option value="">--Seleccionar--</option>';
        db.forEach(m => s.innerHTML += `<option value="${m.id}">${m.n}</option>`);
    });
}

/* 🔧 FIX INSUMOS */
function sumarConsumible() {
    const id = document.getElementById('sel_cons').value;
    const cant = parseFloat(document.getElementById('cons_cant').value) || 1;
    const m = db.find(x => x.id == id);
    if(m){ carritoConsumibles.push({ ...m, c: cant }); renderBadges(); }
}

function renderBadges() {
    const c = document.getElementById('lista_badges_cons');
    c.innerHTML = "";
    carritoConsumibles.forEach(x => {
        c.innerHTML += `<span style="background:#333;color:#b08d57;padding:4px 8px;border-radius:4px;margin-right:4px;">${x.c}x ${x.n}</span>`;
    });
}

/* 🔧 FIX INSUMOS PVC */
function sumarConsumiblePVC(){
    const id = document.getElementById('pv_cons').value;
    const cant = parseFloat(document.getElementById('pv_cant').value) || 1;
    const m = db.find(x => x.id == id);
    if(m){ carritoPVC.push({ ...m, c: cant }); renderPVC(); }
}

function renderPVC(){
    const c = document.getElementById('pv_badges');
    c.innerHTML = "";
    carritoPVC.forEach(x => {
        c.innerHTML += `<span style="background:#333;color:#b08d57;padding:4px 8px;border-radius:4px;margin-right:4px;">${x.c}x ${x.n}</span>`;
    });
}

actualizarSelects();

</script>
