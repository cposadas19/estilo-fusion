<script>
let carrito = [];

function openTab(evt, name) {
    let card = document.getElementsByClassName("card");
    for (let i = 0; i < card.length; i++) card[i].style.display = "none";
    let tab = document.getElementsByClassName("tab");
    for (let i = 0; i < tab.length; i++) tab[i].className = tab[i].className.replace(" active", "");
    document.getElementById(name).style.display = "block";
    evt.currentTarget.className += " active";
}

function agregarPVC() {
    const a = parseFloat(document.getElementById('p_ancho').value);
    const h = parseFloat(document.getElementById('p_alto').value);
    const pTira = parseFloat(document.getElementById('p_tira').value);
    const pVidrio = parseFloat(document.getElementById('p_v_precio').value);
    
    // Lógica técnica de cortes PVC:
    // Carril = Perímetro (2a + 2h)
    // Traslape/Gancho = Depende de altura (2h mínimo por hoja)
    const pulgCarril = (a * 2) + (h * 2);
    const pulgHojas = (h * 4); // 2 traslapes + 2 ganchos
    const m2Vidrio = (a * h) / 1550;
    
    const costo = ((pulgCarril + pulgHojas) * 0.0254 * (pTira/6)) + (m2Vidrio * pVidrio) + 450; // 450 herrajes y felpa
    
    carrito.push({ tipo: 'PVC', desc: document.getElementById('p_desc').value || 'Ventana', costo: costo, det: `${a}x${h}"` });
    render();
}

function agregarMetal() {
    const a = parseFloat(document.getElementById('m_ancho').value);
    const h = parseFloat(document.getElementById('m_alto').value);
    const costo = ((a*2)+(h*2)) * 30; // 30 Lps/pulgada estimado con mano de obra
    carrito.push({ tipo: 'METAL', desc: document.getElementById('m_desc').value || 'Estructura', costo: costo, det: `${a}x${h}"` });
    render();
}

function agregarElec() {
    const c = parseInt(document.getElementById('e_cant').value);
    const p = parseFloat(document.getElementById('e_costo').value);
    carrito.push({ tipo: 'ELEC', desc: document.getElementById('e_desc').value, costo: c * p, det: `${c} pts` });
    render();
}

function render() {
    const lista = document.getElementById('listaItems');
    lista.innerHTML = "";
    let total = 0;
    carrito.forEach((item, i) => {
        total += item.costo;
        lista.innerHTML += `<div class="lista-item"><strong>${item.desc}</strong> (${item.det})<br>L. ${item.costo.toFixed(2)}<button class="btn-del" onclick="carrito.splice(${i},1);render();">X</button></div>`;
    });
    document.getElementById('areaRes').style.display = carrito.length > 0 ? "block" : "none";
    document.getElementById('totalFinal').innerText = "TOTAL: L. " + total.toLocaleString();
}

function enviarWS() {
    let msg = "PROYECTO ESTILO FUSION%0A--------------------%0A";
    let total = 0;
    carrito.forEach(item => {
        msg += `• ${item.desc} (${item.tipo}): L. ${item.costo.toLocaleString()}%0A`;
        total += item.costo;
    });
    msg += `--------------------%0ATOTAL COTIZADO: L. ${total.toLocaleString()}`;
    window.open(`https://wa.me{msg}`);
}
