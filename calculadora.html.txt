    <script>
        const materiales = [
            { nom: "Varilla Lisa 3/8", pre: 165 },
            { nom: "Varilla Lisa 1/2", pre: 275 },
            { nom: "Angulo 1 x 1/8", pre: 175 },
            { nom: "Tubo 1 Cal. 18", pre: 230 }
        ];

        let proyecto = [];

        function showTab(id) {
            document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
            document.querySelectorAll('.tab-link').forEach(l => l.classList.remove('active-tab'));
            document.getElementById(id).classList.add('active');
            document.getElementById('btn-' + (id === 'resumen' ? 'resumen' : id)).classList.add('active-tab');
            if(id === 'resumen') renderResumen();
            actualizarSelectores();
            renderDB();
        }

        function actualizarSelectores() {
            const s = document.querySelector('.select-material');
            s.innerHTML = materiales.map(m => `<option value="${m.pre}">${m.nom}</option>`).join('');
        }

        function addVerjaAlProyecto() {
            const nom = document.getElementById('v-nom').value || "Estructura " + (proyecto.length + 1);
            const ancho = parseFloat(document.getElementById('v-ancho').value);
            const alto = parseFloat(document.getElementById('v-alto').value);
            const sep = parseFloat(document.getElementById('v-sep').value);
            const cant = parseInt(document.getElementById('v-cant').value);
            const preUnidad = parseFloat(document.getElementById('v-mat').value);

            // CÁLCULOS ESPECÍFICOS
            const largoCorte = alto - 2; // Asumiendo perfil de 1" arriba y abajo
            const barrotesXVerja = Math.round((ancho - 2) / (sep + 1));
            const sepFinal = (ancho - 2 - (barrotesXVerja * 1)) / (barrotesXVerja + 1);
            
            proyecto.push({ nom, ancho, alto, largoCorte, barrotesXVerja, cant, sepFinal, preUnidad });
            
            alert("✅ " + nom + " añadida correctamente.");
            showTab('resumen');
        }

        function renderResumen() {
            const lista = document.getElementById('lista-verjas');
            const totales = document.getElementById('totales-proyecto');
            
            if(proyecto.length === 0) {
                lista.innerHTML = "";
                totales.innerHTML = `<p class="text-center text-gray-500 italic text-xs py-10 tracking-widest uppercase">El proyecto está vacío</p>`;
                return;
            }

            // Renderizar cada verja con su "Manual de Taller"
            lista.innerHTML = proyecto.map((v, i) => `
                <div class="bg-gray-50 border-l-4 border-yellow-500 p-4 rounded-xl shadow-sm relative">
                    <button onclick="borrarUno(${i})" class="absolute top-2 right-3 text-red-300 font-bold text-xl">×</button>
                    <p class="font-black text-slate-800 text-sm uppercase">${v.nom} (${v.cant} pzas)</p>
                    <div class="grid grid-cols-2 gap-y-1 mt-2 text-[11px] text-gray-600 font-medium">
                        <p>📏 Medidas: <span class="text-slate-900">${v.ancho}" x ${v.alto}"</span></p>
                        <p>🔪 Corte Barrote: <span class="text-red-600 font-bold">${v.largoCorte}"</span></p>
                        <p>🔢 Barrotes por verja: <span class="text-slate-900 font-bold">${v.barrotesXVerja}</span></p>
                        <p>↔️ Separación: <span class="text-blue-600 font-bold">${v.sepFinal.toFixed(2)}"</span></p>
                    </div>
                </div>
            `).join('');

            // Totales acumulados
            let totalPulgadas = 0;
            let costoTotalBase = 0;
            proyecto.forEach(v => {
                const pulgadasEsteGrupo = (v.barrotesXVerja * v.largoCorte * v.cant);
                totalPulgadas += pulgadasEsteGrupo;
                costoTotalBase += (pulgadasEsteGrupo / 12) * (v.preUnidad / 19.6);
            });

            const totalPies = totalPulgadas / 12;
            const tiras = Math.ceil(totalPies / 19.6);
            const sobrante = (tiras * 19.6) - totalPies;

            totales.innerHTML = `
                <div class="space-y-4">
                    <div class="border-b border-slate-700 pb-4">
                        <p class="text-[10px] text-yellow-500 font-black uppercase tracking-widest">Lista de Compra</p>
                        <p class="text-2xl font-black">Llevar: ${tiras} tiras (6m)</p>
                        <p class="text-[10px] text-green-400 mt-1 uppercase font-bold italic">Sobran ${sobrante.toFixed(2)} pies de la última tira</p>
                    </div>
                    <div class="flex justify-between items-center">
                        <p class="text-[10px] text-gray-400 uppercase font-bold">Costo Material Acumulado</p>
                        <p class="text-2xl font-black text-yellow-400">L ${costoTotalBase.toFixed(2)}</p>
                    </div>
                    <p class="text-[9px] text-gray-500 text-center uppercase tracking-tighter">Suma de todos los cortes: ${totalPies.toFixed(1)} pies lineales</p>
                </div>
            `;
        }

        function renderDB() {
            document.getElementById('tabla-db').innerHTML = materiales.map(m => `
                <div class="flex justify-between p-2 border-b bg-gray-50">
                    <span>${m.nom}</span> <span class="font-bold text-green-700">L ${m.pre}</span>
                </div>
            `).join('');
        }

        function borrarUno(i) { proyecto.splice(i, 1); renderResumen(); }
        function limpiarProyecto() { if(confirm("¿Limpiar todo el proyecto?")) { proyecto = []; renderResumen(); } }
        window.onload = () => { showTab('calc'); };
    </script>
