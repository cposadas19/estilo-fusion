<script>

let proyecto = [];

// detectar pulgadas
function extraer(t){
t = t.toLowerCase();

if(t.includes("/")){
let m = t.match(/(\d+)\/(\d+)/);
return m ? m[1]/m[2] : 1;
}

if(t.includes("x")){
let p = t.split("x");
return Math.max(parseFloat(p[0]), parseFloat(p[1]));
}

let n = t.match(/[\d.]+/);
return n ? parseFloat(n[0]) : 1;
}

// mostrar lista
function render(){

let html="";

proyecto.forEach((v,i)=>{

html += `
<div class="result">
Verja ${i+1}<br>
Barrotes: ${v.info.barrotes}<br>
Separación: ${v.info.separacionReal.toFixed(1)} cm<br>
Altura: ${v.info.altoBarrote.toFixed(1)} cm<br>

<button class="delete" onclick="eliminar(${i})">Eliminar</button>
</div>
`;

});

document.getElementById("lista").innerHTML = html;
}

// agregar verja
function agregar(){

let ancho = parseFloat(document.getElementById("ancho").value);
let alto = parseFloat(document.getElementById("alto").value);
let sep = parseFloat(document.getElementById("sep").value);
let tipo = document.getElementById("tipoBarrote").value;
let modo = document.getElementById("modo").value;
let tuboTexto = document.getElementById("tuboMarco").value;

if(!ancho || !alto || !sep || !tuboTexto){
alert("Faltan datos");
return;
}

let pulg = extraer(tuboTexto);
let tubo = pulg * 2.54;

let division = ancho / sep;
let espacios;

if(modo==="cercano") espacios=Math.round(division);
else if(modo==="menor") espacios=Math.floor(division);
else espacios=Math.ceil(division);

if(espacios<1) espacios=1;

let barrotes = espacios + 1;
let separacionReal = ancho / espacios;
let altoBarrote = alto - (tubo*2);

let marco = [ancho, ancho, alto, alto];
let barrotesArr=[];

for(let i=0;i<barrotes;i++){
barrotesArr.push(altoBarrote);
}

proyecto.push({
marco,
barrotes:barrotesArr,
tipo,
info:{barrotes,separacionReal,altoBarrote}
});

render();
}

// eliminar
function eliminar(i){
proyecto.splice(i,1);
render();
}

// optimizar
function optimizar(piezas,largo){

piezas.sort((a,b)=>b-a);
let barras=[];

piezas.forEach(p=>{

let colocado=false;

for(let i=0;i<barras.length;i++){
if(barras[i]+p<=largo){
barras[i]+=p;
colocado=true;
break;
}
}

if(!colocado){
barras.push(p);
}

});

return barras.length;
}

// calcular
function calcular(){

if(proyecto.length===0){
alert("Agregá verjas");
return;
}

let marcoTotal=[];
let barrotesTotal=[];
let tipo="tubo";

proyecto.forEach(v=>{
marcoTotal=marcoTotal.concat(v.marco);
barrotesTotal=barrotesTotal.concat(v.barrotes);
tipo=v.tipo;
});

let tubosMarco=optimizar(marcoTotal,610);
let largo=(tipo==="varilla")?914:610;
let tubosBarrotes=optimizar(barrotesTotal,largo);

document.getElementById("res").innerHTML=`
Marco: ${tubosMarco} tubos<br>
Barrotes: ${tubosBarrotes} barras<br><br>
TOTAL: ${tubosMarco+tubosBarrotes}
`;
}
