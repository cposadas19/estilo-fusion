function resetDB(){
localStorage.clear();

}

function renderSel(){
let m=document.getElementById('marco');

function guardar(){

function renderDB(){
let l=document.getElementById('listaDB'); l.innerHTML="";

}

function agregar(){

let a=+ancho.value, h=+alto.value;
let sep=+sep.value, gr=+grosor.value;

let m=db.find(x=>x.id==marco.value);
let b=db.find(x=>x.id==barrote.value);

let cant=Math.round(a/sep)-1;
let largo=h-(gr*2);

proyectos.push({
desc:desc.value,
cant,largo,
meta:+meta.value,
ayudante:+ayudante.value,
gan:+gan.value,
mo:+mo.value,
marco:m,bar:b,
usarLamina:usarLamina.checked,
area:(a*h)/144
});

render();
}

function render(){

let sobrantes={};
let total=0, totalSin=0;
let html="", fact="";

proyectos.forEach((p,i)=>{

let costo=0;

[p.marco,p.bar].forEach(mat=>{
if(!mat) return;

let pies=(p.cant*p.largo)/12;

if(!sobrantes[mat.id]) sobrantes[mat.id]=0;

let sin=Math.ceil(pies/mat.u)*mat.p;
totalSin+=sin;

if(sobrantes[mat.id]>=pies){
sobrantes[mat.id]-=pies;
pies=0;
}else{
pies-=sobrantes[mat.id];
sobrantes[mat.id]=0;
}

if(pies>0){
let perf=Math.ceil(pies/mat.u);
costo+=perf*mat.p;
sobrantes[mat.id]+=(perf*mat.u)-pies;
}

});

// LAMINA
if(p.usarLamina){
let lam=db.find(x=>x.tipo==="lamina");
let piezas=Math.ceil(p.area/32);
costo+=piezas*lam.p;
totalSin+=piezas*lam.p;
}

let venta=costo+(costo*(p.gan/100))+p.mo;
let diaria=(venta*p.meta)-p.ayudante;

total+=venta;

html+=`
<div class="item">
<b>${p.desc}</b><br>
Barrotes: ${p.cant} | Largo: ${p.largo.toFixed(2)}<br>
Meta: ${p.meta}<br>
💰 Diario: L ${diaria.toFixed(2)}
</div>
`;

fact+=`${p.desc} - L ${venta.toFixed(2)}<br>`;

});

document.getElementById('lista').innerHTML=html;

document.getElementById('resumen').style.display="block";
document.getElementById('detalle').innerHTML=`
Costo sin optimizar: L ${totalSin.toFixed(2)}<br>
Venta: L ${total.toFixed(2)}<br>
Ahorro: L ${(totalSin-total).toFixed(2)}
`;

document.getElementById('total').innerText="TOTAL: L "+total.toFixed(2);

document.getElementById('factura').innerHTML=fact;

}
