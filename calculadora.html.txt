<script>

let proyecto = [];

// detectar pulgadas
function extraer(t){
t=t.toLowerCase();

if(t.includes("/")){
let m=t.match(/(\d+)\/(\d+)/);
return m?m[1]/m[2]:1;
}

if(t.includes("x")){
let p=t.split("x");
return Math.max(parseFloat(p[0]),parseFloat(p[1]));
}

let n=t.match(/[\d.]+/);
return n?parseFloat(n[0]):1;
}

// agregar verja
function agregar(){

let ancho=parseFloat(document.getElementById("ancho").value);
let alto=parseFloat(document.getElementById("alto").value);
let sep=parseFloat(document.getElementById("sep").value);
let tipo=document.getElementById("tipoBarrote").value;
let tuboTexto=document.getElementById("tuboMarco").value;

if(!ancho||!alto||!sep||!tuboTexto){
alert("Faltan datos");
return;
}

let pulg=extraer(tuboTexto);
let tubo=pulg*2.54;

let espacios=Math.floor(ancho/sep);
if(espacios<1) espacios=1;

let barrotes=espacios+1;
let altoBarrote=alto-(tubo*2);

let piezas=[];

// marco
piezas.push(ancho,ancho,alto,alto);

// barrotes
for(let i=0;i<barrotes;i++){
piezas.push(altoBarrote);
}

proyecto.push({piezas,tipo});

alert("Verja agregada");
}

// calcular optimización
function calcular(){

let piezas=[];
let tubos=[];

proyecto.forEach(v=>{

let largo=(v.tipo==="varilla")?914:610;

v.piezas.forEach(c=>{

let colocado=false;

for(let t of tubos){
if(t.usado+c<=largo){
t.usado+=c;
colocado=true;
break;
}
}

if(!colocado){
tubos.push({usado:c,largo});
}

});

});

let sobrante=0;

tubos.forEach(t=>{
sobrante+=(t.largo-t.usado);
});

res.innerHTML=`
Verjas: ${proyecto.length}<br>
Tubos necesarios: ${tubos.length}<br>
Sobrante total: ${sobrante.toFixed(0)} cm
`;

}

// guardar proyecto
function guardar(){

let nombre=document.getElementById("nombreProyecto").value;

if(!nombre){
alert("Poné nombre al proyecto");
return;
}

let data=JSON.parse(localStorage.getItem("proyectos"))||[];

data.push({nombre,proyecto});

localStorage.setItem("proyectos",JSON.stringify(data));

proyecto=[];
mostrar();
}

// mostrar
function mostrar(){

let data=JSON.parse(localStorage.getItem("proyectos"))||[];
let html="";

data.forEach((p,i)=>{

html+=`
<div class="result">
<b>${p.nombre}</b><br>
Verjas: ${p.proyecto.length}<br>

<input placeholder="Precio cobrado" onchange="ganancia(this)">
<div></div>

<button onclick="eliminar(${i})">Eliminar</button>
</div>
`;
});

lista.innerHTML=html;
}

// ganancia
function ganancia(input){
let precio=parseFloat(input.value);
input.nextElementSibling.innerHTML="Registrado: L "+precio;
}

// eliminar
function eliminar(i){
let data=JSON.parse(localStorage.getItem("proyectos"));
data.splice(i,1);
localStorage.setItem("proyectos",JSON.stringify(data));
mostrar();
}

mostrar();
