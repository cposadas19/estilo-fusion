<script>

let db = JSON.parse(localStorage.getItem('fusion_db'))||[];
let proyecto=[], carritoConsumibles=[], carritoPVC=[];

function openTab(e,n){
document.querySelectorAll('.card').forEach(x=>x.style.display='none');
document.querySelectorAll('.tab').forEach(x=>x.classList.remove('active'));
document.getElementById(n).style.display='block';
e.currentTarget.classList.add('active');
}

function guardarDB(){
let n=db_n.value,p=parseFloat(db_p.value);
if(!n||isNaN(p))return;
db.push({id:Date.now(),n,p});
localStorage.setItem('fusion_db',JSON.stringify(db));
actualizarSelects();
}

function actualizarSelects(){
['sel_marco','sel_barrote','sel_cons','pv_marco','pv_hoja','pv_cons'].forEach(id=>{
let s=document.getElementById(id);
if(!s)return;
s.innerHTML='<option>--</option>';
db.forEach(m=>s.innerHTML+=`<option value="${m.id}">${m.n}</option>`);
});
}

function sumarConsumible(){
let m=db.find(x=>x.id==sel_cons.value);
if(m)carritoConsumibles.push({...m,c:cons_cant.value});
}

function sumarConsumiblePVC(){
let m=db.find(x=>x.id==pv_cons.value);
if(m)carritoPVC.push({...m,c:pv_cant.value});
}

function agregarAlProyecto(){
let m=db.find(x=>x.id==sel_marco.value);
if(!m)return;

let a=p_ancho.value,h=p_alto.value,gr=p_grosor.value,sep=p_sep.value;

let cant=Math.round(a/sep)-1;
let corte=h-(gr*2);

proyecto.push({
desc:p_desc.value,
mats:[
{id:m.id,n:m.n,p:m.p,u:p_unit_m.value,pies:((a*2)+(h*2))/12}
],
cons:[...carritoConsumibles],
info:`${cant} barrotes`
});

renderProyecto();
}

function agregarPVC(){
let m=db.find(x=>x.id==pv_marco.value);
if(!m)return;

let a=pv_ancho.value,h=pv_alto.value;

proyecto.push({
desc:pv_desc.value,
mats:[
{id:m.id,n:m.n,p:m.p,u:pv_unit_m.value,pies:((a*2)+(h*2))/12}
],
cons:[...carritoPVC],
info:"PVC"
});

renderProyecto();
}

function renderProyecto(){
let l=document.getElementById('lista_proyecto_visual');
l.innerHTML="";
proyecto.forEach(p=>{
l.innerHTML+=`<div class="lista-item">${p.desc}</div>`;
});
}

actualizarSelects();

</script>
