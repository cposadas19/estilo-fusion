// NAV
function openMain(e,id){
document.querySelectorAll('.card').forEach(x=>x.classList.remove('active'));
document.querySelectorAll('.tab').forEach(x=>x.classList.remove('active'));
document.getElementById(id).classList.add('active');
e.target.classList.add('active');
}

// BASE
function guardar(){
db.push({id:Date.now(),n:n.value,p:+p.value,tipo:tipo.value});
localStorage.setItem('db',JSON.stringify(db));
renderDB(); renderSelects();
}

function renderDB(){
let html="";
db.forEach((m,i)=>{
html+=`<div class="item">${m.n} - L ${m.p}</div>`;
});
listaDB.innerHTML=html;
}

// SELECTS
function renderSelects(){
mat.innerHTML="";
cons.innerHTML="";
db.forEach(m=>{
if(m.tipo=="material") mat.innerHTML+=`<option value="${m.id}">${m.n}</option>`;
if(m.tipo=="consumible") cons.innerHTML+=`<option value="${m.id}">${m.n}</option>`;
});
}

// CARRITO
function addCons(){
let m=db.find(x=>x.id==cons.value);
let c=+cant.value;
carrito.push({n:m.n,p:m.p,c});
renderCarrito();
}

function renderCarrito(){
let html="";
carrito.forEach(x=>{
html+=`<span class="badge">${x.c}x ${x.n}</span>`;
});
carrito.innerHTML=html;
}

function totalCons(){
let t=0;
carrito.forEach(x=>t+=x.p*x.c);
return t;
}

// PROYECTO
function addProyecto(){

render();
}

});

total.innerHTML="TOTAL: L "+t;
}
