<script>

let totalProyecto = 0;
let cantidad = 0;

function agregar(){

let ancho = parseFloat(document.getElementById("ancho").value);
let alto = parseFloat(document.getElementById("alto").value);
let sep = parseFloat(document.getElementById("sep").value);

let precioTubo = parseFloat(document.getElementById("tubo").value);
let precioVarilla = parseFloat(document.getElementById("varilla").value);
let precioTuboBarrote = parseFloat(document.getElementById("tuboBarrote").value);
let precioPlatina = parseFloat(document.getElementById("platina").value);
let lamina = parseFloat(document.getElementById("lamina").value) || 0;
let produccion = parseFloat(document.getElementById("produccion").value) || 1;

if(!ancho || !alto || !sep){
alert("Faltan datos");
return;
}

// BARROTES
let barrotes = Math.round(ancho/sep)+1;

// MARCO
let tuboTotal = (2*ancho + 2*alto)/600;
let costoTubo = Math.ceil(tuboTotal)*precioTubo;

// VARILLA
let varillaTotal = (barrotes*(alto-10))/914;
let costoVarilla = Math.ceil(varillaTotal)*precioVarilla;

// OTROS MATERIALES
let costoTuboBarrote = Math.ceil(varillaTotal)*precioTuboBarrote;
let costoPlatina = Math.ceil(varillaTotal)*precioPlatina;

// MATERIAL TOTAL
let material = costoTubo + costoVarilla + costoTuboBarrote + costoPlatina + lamina;

// MANO DE OBRA POR PRODUCCIÓN
let manoTotalDia = 1200;
let mano = manoTotalDia / produccion;

// COSTO TOTAL
let costo = material + mano;

// PRECIOS
let precioPro = costo * 1.7;
let precioRapido = material * 2.5;

// ACUMULADO
totalProyecto += precioPro;
cantidad++;

let promedio = totalProyecto / cantidad;

// MOSTRAR
document.getElementById("resultado").innerHTML += `
<hr>
Verja ${cantidad}<br>
Barrotes: ${barrotes}<br><br>

Material: L ${material.toFixed(0)}<br>
Mano de obra: L ${mano.toFixed(0)}<br>
Costo total: L ${costo.toFixed(0)}<br><br>

Precio PRO: L ${precioPro.toFixed(0)}<br>
Precio rápido: L ${precioRapido.toFixed(0)}<br><br>

TOTAL PROYECTO: L ${totalProyecto.toFixed(0)}<br>
PROMEDIO: L ${promedio.toFixed(0)}<br>
`;
}

</script>
