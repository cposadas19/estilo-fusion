function optimizarPiezas(piezas,largo){

piezas.sort((a,b)=>b-a);

let barras = [];

piezas.forEach(pieza=>{

let mejorBarra = -1;
let menorSobrante = Infinity;

for(let i=0;i<barras.length;i++){

if(barras[i] >= pieza){

let sobrante = barras[i] - pieza;

if(sobrante < menorSobrante){

menorSobrante = sobrante;
mejorBarra = i;

}

}

}

if(mejorBarra >= 0){

barras[mejorBarra] -= pieza;

}else{

barras.push(largo - pieza);

}

});

return {

cantidad:barras.length,

sobrantes:barras.filter(x=>x>0)

};

}
